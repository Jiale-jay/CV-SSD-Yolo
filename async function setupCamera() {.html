async function setupCamera() { const video = document.getElementById('video'); const stream = await navigator.mediaDevices.getUserMedia({ video: {facingMode: 'user'} }); video.srcObject = stream; return new Promise((resolve) => { video.onloadedmetadata = () => { resolve(video); }; }); } async function loadModel() { // 从 cocoSsd 模块加载 SSD 模型 return await cocoSsd.load(); } async function detectObjects(video, model) { const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d'); video.width = video.videoWidth; video.height = video.videoHeight; canvas.width = video.width; canvas.height = video.height; async function frame() { // 清除画布 ctx.clearRect(0, 0, canvas.width, canvas.height); // 使用模型进行目标检测 const predictions = await model.detect(video); // 绘制检测到的物体 predictions.forEach(prediction => { // 绘制边界框 ctx.beginPath(); ctx.rect(...prediction.bbox); ctx.lineWidth = 2; ctx.strokeStyle = 'red'; ctx.stroke(); // 绘制标签 ctx.fillStyle = 'red'; ctx.fillText( `${prediction.class} (${Math.round(prediction.score * 100)}%)`, prediction.bbox[0], prediction.bbox[1] > 10 ? prediction.bbox[1] - 5 : 10 ); }); // 继续进行下一帧的检测 requestAnimationFrame(frame); } // 开始检测循环 frame(); } async function main() { const video = await setupCamera(); video.play(); const model = await loadModel(); detectObjects(video, model); } main();
